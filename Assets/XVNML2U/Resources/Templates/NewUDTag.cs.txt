using XVNML.Core.Tags;

[AssociateWithTag("udtag", typeof(#SCRIPTNAME#), TagOccurance.Multiple, true)]
public sealed class UDTag#SCRIPTNAME# : TagBase
{
	protected override string[] AllowedParameters => new[]
	{
		"param1",
		"param2",
		"param3"
	}
	
	// Uncomment this to set flags for user-defined tag.
	//protected override string[] AllowedFlags => new[]
	//{
	//	"flag1"
	//	"flag2"
	//	"flag3"
	//}
	
	// Create your tag members here...
	public string param1;
	public int param2;
	public bool param3;
	
	public override void OnResolve(string fileOrigin)
	{
		// Always call this, or default property for tag will not function
		base.OnResolve(fileOrigin);
		
		int paramElementID = 0;
		
		param1 = GetParameterValue<string>(AllowedParameters[paramElementID++]);
		param2 = GetParameterValue<int>(AllowedParameters[paramElementID++]);
		param3 = GetParameterValue<bool)(AllowedParameters[paramElementID]);
	}
}
